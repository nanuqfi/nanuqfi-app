'use client'

import { type ReactNode } from 'react'
import type { RiskLevel } from '@/lib/mock-data'

const tierBorderColors: Record<RiskLevel, string> = {

interface GlassCardProps {
  children: ReactNode
  tier?: RiskLevel
  elevated?: boolean
  className?: string
}

export function GlassCard({ children, tier, elevated, className = '' }: GlassCardProps) {
  const base = 'glass rounded-2xl relative overflow-hidden'
  const tint = tier ? `border-l-2 ${tierBorderColors[tier]}` : ''
  const elevation = elevated ? 'glass-elevated' : ''

  return (
    <div className={`${base} ${tint} ${elevation} ${className}`}>
      {children}
    </div>
  )
}

// src/components/ui/animated-counter.tsx
'use client'

import { useEffect, useRef, useState } from 'react'

interface AnimatedCounterProps {
  end: number
  duration?: number
  prefix?: string
  suffix?: string
  decimals?: number
  className?: string
}

export function AnimatedCounter({ end, duration = 1500, prefix = '', suffix = '', decimals = 0, className = '' }: AnimatedCounterProps) {
  const [count, setCount] = useState(0)
  const ref = useRef<HTMLSpanElement>(null)

  useEffect(() => {
    let start = 0
    const step = end / (duration / 16)
    const timer = setInterval(() => {
      start += step
      if (start >= end) {
        setCount(end)
        clearInterval(timer)
      } else {
        setCount(start)
      }
    }, 16)
    return () => clearInterval(timer)
  }, [end, duration])

  return (
    <span ref={ref} className={`font-mono tabular-nums ${className}`}>
      {prefix}{count.toFixed(decimals)}{suffix}
    </span>
  )
}

import { type ReactNode, useRef } from 'react'
import { useInView } from '@/hooks/use-in-view'

interface FadeInProps {
  children: ReactNode
  delay?: number
  className?: string
}

export function FadeIn({ children, delay = 0, className = '' }: FadeInProps) {
  const ref = useRef<HTMLDivElement>(null)
  const inView = useInView(ref, { threshold: 0.1 })

  return (
    <div
      ref={ref}
      className={`transition-all duration-700 ease-out ${className}`}
      style={{
        opacity: inView ? 1 : 0,
        transform: inView ? 'translateY(0)' : 'translateY(30px)',
        transitionDelay: `${delay}ms`,
      }}
    >
      {children}
    </div>
  )
}

'use client'

import { useState } from 'react'

const ranges = ['1W', '1M', '3M', '1Y', 'All'] as const
type TimeRange = (typeof ranges)[number]

interface TimeRangeSelectorProps {
  defaultRange?: TimeRange
  onChange?: (range: TimeRange) => void
  className?: string
}

export function TimeRangeSelector({ defaultRange = '1M', onChange, className = '' }: TimeRangeSelectorProps) {
  const [active, setActive] = useState<TimeRange>(defaultRange)

  const handleSelect = (range: TimeRange) => {
    setActive(range)
    onChange?.(range)
  }

  return (
    <div className={`flex gap-1 bg-slate-900/60 p-1 rounded-xl border border-white/5 ${className}`}>
      {ranges.map((r) => (
        <button
          key={r}
          onClick={() => handleSelect(r)}
          className={`px-3 py-1.5 rounded-lg text-xs font-mono font-medium transition-colors ${
            active === r
              ? 'bg-slate-700 text-white shadow-sm ring-1 ring-white/10'
              : 'text-slate-400 hover:text-white hover:bg-slate-800'
          }`}
        >
          {r}
        </button>
      ))}
    </div>
  )
}

'use client'

import Link from 'next/link'
import { usePathname } from 'next/navigation'
import { useWallet } from '@solana/wallet-adapter-react'
import { useWalletModal } from '@solana/wallet-adapter-react-ui'
import { LayoutDashboard, Vault, Activity } from 'lucide-react'
import { SystemStatus } from './system-status'

const navLinks = [
  { href: '/app', label: 'Dashboard', icon: LayoutDashboard },
  { href: '/app/vaults', label: 'Vaults', icon: Vault },
  { href: '/app/activity', label: 'Activity', icon: Activity },
]

export function Nav() {
  const pathname = usePathname()
  const { publicKey, disconnect } = useWallet()
  const { setVisible } = useWalletModal()

  const isActive = (href: string) => {
    if (href === '/app') return pathname === '/app'
    return pathname.startsWith(href)
  }

  const truncatedKey = publicKey
    ? `${publicKey.toBase58().slice(0, 4)}...${publicKey.toBase58().slice(-4)}`
    : null

  return (
    <nav className="fixed top-0 left-0 right-0 h-16 z-50 bg-slate-900/80 backdrop-blur-xl border-b border-white/5 flex items-center justify-between px-6 lg:px-8">

          <span className="text-white font-bold text-sm">N</span>
        </div>
        <span className="font-semibold text-lg tracking-tight text-white">
          Nanuq<span className="text-sky-500">Fi</span>
        </span>
      </div>

      {/* Center Nav */}
      <div className="hidden md:flex items-center justify-center w-2/4">
        <div className="flex gap-1 bg-slate-900/50 p-1 rounded-full border border-white/5">

            <Link
              key={href}
              href={href}
              className={`px-5 py-1.5 rounded-full text-sm font-medium transition-all duration-200 ${

      {/* Right */}
      <div className="flex items-center justify-end gap-4 w-1/4">
        <SystemStatus />
        {truncatedKey ? (
          <button
            onClick={() => disconnect()}
            className="flex items-center gap-2 px-4 py-1.5 rounded-lg bg-white/5 border border-white/10 hover:bg-white/10 transition-all"

            <div className="w-2 h-2 rounded-full bg-emerald-500 animate-pulse" />

import { useKeeperHealth, useVaultData } from '@/hooks/use-keeper-api'
import { formatUsd, formatApy, getTotalTvl, getWeightedApy, mockVaults } from '@/lib/mock-data'

export function PortfolioSummary() {
  const { data: health } = useKeeperHealth()
  const { data: modVault } = useVaultData('moderate')
  const { data: aggVault } = useVaultData('aggressive')

  const tvl = modVault && aggVault
    ? (modVault.tvl ?? 0) + (aggVault.tvl ?? 0)
    : getTotalTvl(mockVaults)

  const apy = modVault && aggVault
    ? ((modVault.apy ?? 0) + (aggVault.apy ?? 0)) / 2
    : getWeightedApy(mockVaults)

  const dailyEarnings = tvl * (apy / 100) / 365

  const lastActive = health?.lastCycle
    ? `${Math.round((Date.now() - new Date(health.lastCycle).getTime()) / 60000)}m ago`
    : '—'

  return (
    <GlassCard className="p-6 shadow-2xl shadow-black/40">
      <div className="grid grid-cols-2 lg:grid-cols-4 gap-6 sm:divide-x divide-white/5">
        <div className="flex flex-col gap-1 sm:px-4 first:pl-0">
          <span className="text-[11px] uppercase tracking-wider text-slate-400 font-semibold">Total Value</span>
          <span className="text-2xl font-mono tabular-nums text-white">{formatUsd(tvl)}</span>
